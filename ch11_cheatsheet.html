import syntax
IMPORT All Import Forms
# Import entire module
import math
print(math.sqrt(16))       # 4.0

# Import with alias
import numpy as np
import pandas as pd

# Import specific names
from math import sqrt, pi
print(sqrt(16))             # 4.0 — no math. prefix needed

# Import with alias
from math import sqrt as sq
print(sq(25))               # 5.0

# Import everything — avoid in production
from math import *
`from module import *` pollutes namespace — avoid unless module defines `__all__`
creating your own module
CREATE mymodule.py → import it
# File: mymodule.py
greeting = "Hello from module"

def add(a, b):
    return a + b

def greet(name):
    return f"Hello, {name}"

# File: main.py — import and use
import mymodule

print(mymodule.greeting)
print(mymodule.add(3, 4))     # 7
print(mymodule.greet("Bob"))  # Hello, Bob
Rule: Module file must be in the same directory (or on sys.path) to be importable
__name__ == "__main__"
MAIN Why & How
# File: mymodule.py
def add(a, b):
    return a + b

if __name__ == "__main__":
    # Only runs when file is executed directly
    # NOT when imported by another module
    print(add(3, 4))   # 7
__name__: When run directly → `"__main__"`; when imported → module's filename (e.g. `"mymodule"`)
MAIN Real World Use
# utils.py
def clean_data(data):
    return [x.strip() for x in data]

if __name__ == "__main__":
    # Test the function when running directly
    sample = [" hello ", " world "]
    print(clean_data(sample))   # ['hello', 'world']

# When imported: clean_data() is available
# but the test block does NOT run
Always wrap test/run code in `if __name__ == "__main__":` to make modules safely importable

PKG Package Structure
# A package is a folder with __init__.py
mypackage/
├── __init__.py        # makes it a package
├── utils.py
└── helpers.py

# Import from package
from mypackage import utils
from mypackage.utils import clean_data
import mypackage.helpers as h
__init__.py: Can be empty or define what's exported with __all__

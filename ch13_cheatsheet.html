loads & dumps — strings
PARSE json.loads() — string → Python
import json

json_str = '{"name": "Alice", "age": 25}'

data = json.loads(json_str)   # parse string → dict

print(data)           # {'name': 'Alice', 'age': 25}
print(data["name"])   # Alice
print(type(data))     # <class 'dict'>
loads = load string. Input must be a valid JSON string.
DUMP json.dumps() — Python → string
import json

data = {"name": "Alice", "age": 25, "active": True}

json_str = json.dumps(data)
print(json_str)
# {"name": "Alice", "age": 25, "active": true}
print(type(json_str))  # <class 'str'>
dumps = dump to string. Notice Python True → JSON true automatically.
DUMP dumps() — Pretty Print with indent
data = {"name": "Alice", "scores": [90, 85, 92]}

pretty = json.dumps(data, indent=4)
print(pretty)
# {
#     "name": "Alice",
#     "scores": [90, 85, 92]
# }

# sort keys alphabetically
json.dumps(data, indent=4, sort_keys=True)
load & dump — files
FILE json.load() — read from file
import json

with open("data.json", "r") as f:
    data = json.load(f)

print(data)       # dict / list
print(data["name"])
load (no s) = reads from a file object.
FILE json.dump() — write to file
import json

data = {"name": "Alice", "age": 25}

with open("data.json", "w") as f:
    json.dump(data, f, indent=4)

# data.json now contains pretty JSON
dump (no s) = writes to a file object.

working with nested json
PARSE Accessing Nested Data
json_str = '''
{
  "user": {
    "name": "Alice",
    "scores": [90, 85, 92],
    "address": {"city": "Delhi", "pin": "110001"}
  }
}
'''

data = json.loads(json_str)

print(data["user"]["name"])              # Alice
print(data["user"]["scores"][0])          # 90
print(data["user"]["address"]["city"])   # Delhi
PARSE Looping over JSON array
json_str = '[{"name":"Alice","age":25},{"name":"Bob","age":30}]'

users = json.loads(json_str)

for user in users:
    print(user["name"], user["age"])
# Alice 25
# Bob 30
common errors & traps
TIP JSONDecodeError — Invalid JSON
# Single quotes are NOT valid JSON
bad = "{'name': 'Alice'}"
json.loads(bad)   # JSONDecodeError!

# Must use double quotes
good = '{"name": "Alice"}'
json.loads(good)  # works fine
JSON requires double quotes. Single quotes cause JSONDecodeError.
TIP Non-serialisable Types
import json
from datetime import datetime

data = {"now": datetime.now()}
json.dumps(data)   # TypeError: not JSON serializable

# Fix: convert to string first
data = {"now": str(datetime.now())}
json.dumps(data)   # works
datetime, set, and custom objects are not JSON serialisable by default — convert to str / list first.
TIP Trailing commas invalid
# INVALID JSON
{
  "name": "Alice",   ← trailing comma
}

# VALID JSON
{
  "name": "Alice"
}
TIP Safe reading with .get()
data = {"name": "Alice"}

# KeyError if key missing
data["age"]           # KeyError!

# Safe with default
data.get("age", 0)   # 0

Chapter 4 — Quick Revision Cheat Sheet

defining functions
DEF Basic Syntax
def greet():
    print("Hello")

greet()    # Hello
Rule: def keyword, function name, parentheses, colon — body indented
DEF Parameters & Default Values
def greet(name, message="Hello"):
    print(f"{message}, {name}")

greet("Aditya")           # Hello, Aditya
greet("Aditya", "Hi")    # Hi, Aditya
Rule: Default parameters must always come AFTER non-default parameters
RETURN Returning Values
def add(a, b):
    return a + b

result = add(3, 4)   # 7

# Return multiple values (returns a tuple)
def min_max(nums):
    return min(nums), max(nums)

lo, hi = min_max([3, 1, 7])   # lo=1, hi=7
Key: A function with no return statement returns None implicitly
positional & keyword arguments
ARGS Positional vs Keyword
def info(name, age, city):
    print(name, age, city)

# Positional — order matters
info("Aditya", 25, "Gurgaon")

# Keyword — order doesn't matter
info(age=25, city="Gurgaon", name="Aditya")

# Mix — positional first, keyword after
info("Aditya", city="Gurgaon", age=25)
*args — variable positional arguments
*ARGS Syntax & Usage
def total(*numbers):
    print(numbers)          # tuple: (1, 2, 3, 4)
    return sum(numbers)

total(1, 2, 3, 4)         # 10
total(5, 10)               # 15
Key: *args collects extra positional arguments into a tuple
*ARGS Mixed with regular params
def log(level, *messages):
    for msg in messages:
        print(f"[{level}] {msg}")

log("INFO", "Started", "Loading", "Done")
# [INFO] Started
# [INFO] Loading
# [INFO] Done
Rule: *args must come after regular positional params
**kwargs — variable keyword arguments
**KWARGS Syntax & Usage
def profile(**details):
    print(details)           # dict: {'name': 'Aditya', 'age': 25}
    for k, v in details.items():
        print(f"{k}: {v}")

profile(name="Aditya", age=25, city="Gurgaon")
Key: **kwargs collects extra keyword arguments into a dict
**KWARGS Full parameter order
def func(a, b, *args, **kwargs):
    print(a, b)        # regular params
    print(args)        # extra positional → tuple
    print(kwargs)      # extra keyword → dict

func(1, 2, 3, 4, x=5, y=6)
# 1 2
# (3, 4)
# {'x': 5, 'y': 6}
Order rule: regular → *args → keyword-only → **kwargs
UNPACK Unpacking with * and **
nums = [1, 2, 3]
print(*nums)           # 1 2 3  (unpacks list)

def add(a, b, c): return a + b + c
add(*nums)             # 6 — unpacks into positional args

data = {"name": "Aditya", "age": 25}
profile(**data)        # unpacks dict into keyword args
scope — LEGB rule
SCOPE LEGB — Python lookup order
LetterScopeExample
LLocal — inside current functionx inside def f()
EEnclosing — outer function (nested)x in outer def
GGlobal — module levelx at top of file
BBuilt-in — Python builtinsprint, len, range
Rule: Python searches L → E → G → B — stops at first match
GLOBAL global keyword
x = 10          # global

def modify():
    global x
    x = 99         # modifies global x

modify()
print(x)          # 99
Avoid global — makes code harder to test and debug
NONLOCAL nonlocal keyword
def outer():
    x = 10

    def inner():
        nonlocal x
        x = 99    # modifies outer's x

    inner()
    print(x)      # 99

outer()
nonlocal modifies enclosing (E) scope — used in closures and nested functions
lambda — anonymous functions
LAMBDA Syntax & Examples
# Syntax: lambda parameters: expression

square = lambda x: x ** 2
print(square(5))         # 25

add = lambda a, b: a + b
print(add(3, 4))          # 7

# Common use — sorting with key
names = ["Aditya", "Bob", "Charlie"]
sorted(names, key=lambda x: len(x))
# ['Bob', 'Aditya', 'Charlie']
Use lambda for: short one-liner functions passed as arguments (sorted, map, filter)
common mistakes
DEBUG Mutable Default Argument Trap
# ✗ WRONG — list is created once and shared
def add_item(item, lst=[]):
    lst.append(item)
    return lst

add_item(1)   # [1]
add_item(2)   # [1, 2]  ← BUG! carries over from last call

# ✓ CORRECT — use None as default
def add_item(item, lst=None):
    if lst is None:
        lst = []
    lst.append(item)
    return lst
Never use mutable objects (list, dict, set) as default parameter values

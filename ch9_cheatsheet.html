file modes
MODES All File Open Modes
ModeNameBehaviour
"r"ReadDefault. Error if file doesn't exist
"w"WriteCreates file. Overwrites if exists
"a"AppendCreates file. Adds to end if exists
"x"CreateCreates file. Error if already exists
"r+"Read + WriteFile must exist
"w+"Write + ReadOverwrites existing
"rb"Read BinaryFor images, PDFs, etc.
"wb"Write BinaryFor binary output
"w" mode silently overwrites the entire file — use "a" to append instead
open() & with statement
OPEN open() vs with statement
# Old way — must manually close
f = open("data.txt", "r")
content = f.read()
f.close()              # easy to forget!

# Correct way — with auto-closes file
with open("data.txt", "r") as f:
    content = f.read()
# file is automatically closed here
Always use `with open()` — file closes automatically even if an error occurs
WITH Opening with encoding
with open("data.txt", "r", encoding="utf-8") as f:
    content = f.read()
Best practice: Always specify encoding="utf-8" to avoid platform-specific issues
reading files
READ All Read Methods
with open("data.txt", "r") as f:

    content = f.read()        # entire file as one string
    line    = f.readline()    # one line at a time (includes \n)
    lines   = f.readlines()   # all lines as a list of strings
readline() includes the \n at the end — use .strip() to remove it
READ Looping line by line
# Most memory-efficient — reads one line at a time
with open("data.txt", "r") as f:
    for line in f:
        print(line.strip())   # strip() removes \n

# Read all lines, strip each
with open("data.txt") as f:
    lines = [line.strip() for line in f]
Loop over file object directly for large files — doesn't load entire file into memory
writing files
WRITE write() & writelines()
# write() — overwrites file
with open("data.txt", "w") as f:
    f.write("Hello\n")
    f.write("World\n")

# append() — adds to end of file
with open("data.txt", "a") as f:
    f.write("New line\n")

# writelines() — writes list of strings
lines = ["line1\n", "line2\n", "line3\n"]
with open("data.txt", "w") as f:
    f.writelines(lines)
write() vs writelines(): write() takes a single string; writelines() takes a list of strings — neither adds \n automatically
file pointer
SEEK tell() & seek()
with open("data.txt", "r") as f:
    print(f.tell())         # 0 — current position in bytes
    f.read(5)               # read 5 bytes
    print(f.tell())         # 5

    f.seek(0)               # go back to start
    f.seek(0, 2)            # go to end of file

    content = f.read()      # read again from start
After read(): file pointer is at the end — use seek(0) to read again from start
os module — file & path operations
OS Common os Operations
import os

os.path.exists("data.txt")    # True/False — file exists?
os.path.isfile("data.txt")    # True if it's a file
os.path.isdir("myfolder")     # True if it's a directory
os.path.getsize("data.txt")   # file size in bytes
os.getcwd()                    # current working directory
os.listdir(".")               # list files in directory
os.remove("data.txt")         # delete file
os.rename("old.txt", "new.txt") # rename file
os.makedirs("folder/sub", exist_ok=True)  # create dirs
PATH pathlib — modern path handling
from pathlib import Path

p = Path("data.txt")

p.exists()          # True/False
p.read_text()        # read entire file as string
p.write_text("hi")  # write string to file
p.stem               # "data" — filename without extension
p.suffix             # ".txt" — extension
p.parent             # parent directory
p.name               # "data.txt" — full filename

# Build paths safely (works on Windows & Linux)
full_path = Path("folder") / "sub" / "data.txt"
Prefer pathlib over os.path — cleaner, object-oriented, cross-platform
error handling with files
SAFE Handle missing files gracefully
# Check before opening
if os.path.exists("data.txt"):
    with open("data.txt") as f:
        content = f.read()

# Or use try/except — preferred
try:
    with open("data.txt", "r") as f:
        content = f.read()
except FileNotFoundError:
    print("File not found")
except PermissionError:
    print("No permission to read file")
